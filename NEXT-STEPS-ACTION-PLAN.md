function exportToPDF(path) {
  const doc = new jsPDF();
  doc.text('Learning Path', 10, 10);
  path.forEach((node, i) => {
    doc.text(`${i+1}. ${node.name}`, 10, 20 + i*10);
  });
  doc.save('learning-path.pdf');
}

      'zh-CN': { ... },
      'en-US': { ... }
    };
  }
  
  t(key) {
    return this.messages[this.locale][key];
  }
  
  setLocale(locale) {
    this.locale = locale;
    this.updateUI();
  }
